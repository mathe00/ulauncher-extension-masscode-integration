# Force pure-Python PyYAML to avoid compiled _yaml.so incompatibilities
# across different Python versions (e.g., extension installed with 3.12
# but Ulauncher runs on 3.14). The pure-Python version works everywhere.
--no-binary yaml

pyperclip
fuzzywuzzy
# NOTE: python-Levenshtein is intentionally excluded from this list.
# It installs compiled .so files that can conflict with Ulauncher's own
# Levenshtein dependency if the Python version differs (e.g., extension
# installed with Python 3.12 but Ulauncher runs on Python 3.14).
# fuzzywuzzy works without it (with a performance warning).
pyyaml              # required for V5+ Markdown Vault (.meta.yaml and frontmatter parsing)
